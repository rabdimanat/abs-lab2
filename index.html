      throw new IndexOutOfBoundsException();
      }

      Node currNode = head;
      Node preTargetNode = null;
      Node newNode = new Node(element);
      if (index == 0) {
      newNode.next = head;
      head = newNode;
      } else {

      for (int i = 0; i < index; i++) {
      preTargetNode = currNode;
      currNode = currNode.next;
      }

      preTargetNode.next = (newNode);
      newNode.next = currNode;

      if (preTargetNode == tail) {
      tail = newNode;
      tail.next = null;
      }

      }
      this.size++;
      }
      public int find(T keyItem) {
      Node currnode = this.head;
      int index = 0;
      while(index < this.size()) {
      if(currnode.data.equals(keyItem)) {
      return index;
      }
      index++;
      currnode = currnode.next;
      }
      return -1;
      }
      public T remove(int index) {
      if (index<0 || index >=size()) {
      throw new IndexOutOfBoundsException();
      }
      Node ref = head;
      for (int i = 0; i < index; i++) {
      ref = ref.next;
      }
      if (index == 0) {
      head = ref.next;
      } else {
      ref.prev.next = ref.next;
      }
      size--;
      return (T) ref.data;
      }

      Node reverse()
      {
      Node prev = null;
      Node current = head;
      Node next = null;
      while (current != null) {
      next = current.next;
      current.next = prev;
      prev = current;
      current = next;
      }
      head = prev;
